pipeline {
    agent any
    tools{
        maven 'maven'
    }
    environment {
        registry = "albatros123/test" 
        registryCredential = 'bf3d08a2-c3fc-4e5c-a418-126f625839a3'
        dockerImage = ''
    }
    stages {
        stage('Cloning git') { 
            steps {
                echo 'Clone the repo'
                sh 'rm -fr git-docker-kubernetes'
                git branch: 'master', url: 'https://github.com/logicaldoc/community.git'
            }
        }
        stage('Maven test') {
            steps {
                sh 'mvn -v'
                script {
                    warnError(message: "Maven test stage was unstable.", catchInterruptions: false) {
                    sh 'mvn clean install'
                    sh 'mvn clean package'
                    }
                }
            }
        }
        stage('Building image') { 
            steps {
                script {
                 dockerImage = docker.build registry + ":$BUILD_NUMBER"
                }
            }
        }
        stage('Creating Docker image') { 
            steps {
                script {
                    
                  docker.withRegistry('', registryCredential) { 
                    dockerImage.push() 
                    
                    }
                }
            }
        }
        stage('Deploy to Kubernetes') { 
            steps { 
                echo 'Deploy to Kubernetes'
                }
            } 
        }
        post {
            always {
                sh 'docker logout'
                
            }
        } 
    }
